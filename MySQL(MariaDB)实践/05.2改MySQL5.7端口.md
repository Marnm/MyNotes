---
title: 05.2改MySQL5.7端口
tags: []
---

```shell
[root@jaking ~]# netstat -pantul | grep mysql
tcp6       0      0 :::3306                 :::*                    LISTEN      10305/mysqld 
[root@jaking ~]# cat /etc/my.cnf             
[mysqld]
datadir=/var/lib/mysql
socket=/var/lib/mysql/mysql.sock
symbolic-links=0
port=3307
log-error=/var/log/mysqld.log
pid-file=/var/run/mysqld/mysqld.pid
skip-name-resolve
[root@jaking ~]# systemctl restart mysqld
[root@jaking ~]# netstat -pantul | grep mysql
tcp6       0      0 :::3307                 :::*                    LISTEN      10396/mysqld
```
